import os
import sys
import json
import torch
from torch_geometric.loader import DataLoader
from tqdm import tqdm

# 环境设置
current_dir = os.getcwd()
sys.path.insert(0, current_dir)

from utils.dataset import ZincLMDBDataset

# 常量定义 (必须与生成的 Latent 脚本保持一致以保证对齐)
CHUNK_SIZE = 81920 
BATCH_SIZE = 5120    
OUTPUT_DIR = "/home/course_project/MolFLAE/Latent_Experiments/data/zinc9m_atoms/"
# 为了避免覆盖原来生成 Latent 时的断点记录，这里改用一个新的 JSON
PROGRESS_FILE = os.path.join(OUTPUT_DIR, "metadata_atoms.json")

def load_progress():
    if os.path.exists(PROGRESS_FILE):
        with open(PROGRESS_FILE, 'r') as f:
            return json.load(f)
    return {"last_processed_idx": -1, "chunk_count": 0}

def save_progress(idx, chunk_count):
    with open(PROGRESS_FILE, 'w') as f:
        json.dump({"last_processed_idx": idx, "chunk_count": chunk_count}, f)

def main():
    os.makedirs(OUTPUT_DIR, exist_ok=True)

    print("Loading Training Dataset for Atom Counts Extraction...")
    lmdb_path = "/home/course_project/MolFLAE/Latent_Experiments/data/zinc9m/ZINC_9M.lmdb"
    indices_path = "/home/course_project/MolFLAE/Latent_Experiments/data/zinc9m/dataset_indices.npy"
    
    # 初始化数据集
    dataset = ZincLMDBDataset(lmdb_path, indices_path, split='train')
    
    # 获取断点信息
    progress = load_progress()
    start_idx = progress["last_processed_idx"] + 1
    chunk_count = progress["chunk_count"]
    
    if start_idx >= len(dataset):
        print("All molecules' atom counts already extracted.")
        return

    print(f"Resuming from index: {start_idx}, Current Chunk Count: {chunk_count}")

    # 使用 Subset 实现从断点开始
    subset_indices = list(range(start_idx, len(dataset)))
    subset_dataset = torch.utils.data.Subset(dataset, subset_indices)
    
    # 注意：shuffle=False 非常重要，保证与 latent_dataset 的分子顺序完全一致
    loader = DataLoader(
        subset_dataset, 
        batch_size=BATCH_SIZE, 
        shuffle=False, 
        num_workers=8, 
        pin_memory=False
    )

    atom_buffer =[]
    current_idx_in_total = start_idx

    pbar = tqdm(total=len(dataset), initial=start_idx, desc="Extracting Atom Counts")
    
    for batch in loader:
        num_mols_in_batch = batch.batch.max().item() + 1
        batch_mol_ids = dataset.ids[current_idx_in_total : current_idx_in_total + num_mols_in_batch]
        
        try:
            # 在 PyTorch Geometric 中，batch.ptr 记录了每个图的节点索引边界
            # 相邻两个 ptr 的差值，就是这个图（分子）包含的节点（原子）数
            # 这种提取方式是最快、最准的
            if hasattr(batch, 'ptr') and batch.ptr is not None:
                atom_counts = (batch.ptr[1:] - batch.ptr[:-1]).tolist()
            else:
                # 备用方案：如果由于某些原因没有 ptr 属性
                atom_counts = batch.batch.bincount().tolist()

            for i in range(num_mols_in_batch):
                atom_buffer.append({
                    'id': batch_mol_ids[i],
                    'n': float(atom_counts[i])  # 统一转成 float 方便后续 Regression 训练
                })

        except Exception as e:
            print(f"\n[Warning] CPU Processing batch failed at {current_idx_in_total}: {e}")
            # 原脚本中的对齐补救措施：如果真发生了极端的读取错误，用 0 占位
            # 由于这个脚本只跑 CPU 读取，基本上不会触发这个异常
            for i in range(num_mols_in_batch):
                atom_buffer.append({
                    'id': batch_mol_ids[i],
                    'n': 0.0
                })

        current_idx_in_total += num_mols_in_batch
        pbar.update(num_mols_in_batch)

        # 检查是否满一个 Chunk 并保存
        if len(atom_buffer) >= CHUNK_SIZE:
            chunk_count += 1
            # 命名为 chunk_XXXX_atoms.pt，与 Latent 数据进行区分并一一对应
            chunk_filename = os.path.join(OUTPUT_DIR, f"chunk_{chunk_count:04d}_atoms.pt")
            
            torch.save(atom_buffer[:CHUNK_SIZE], chunk_filename)
            
            save_progress(current_idx_in_total - (len(atom_buffer) - CHUNK_SIZE) - 1, chunk_count)
            
            # 移除已保存部分
            atom_buffer = atom_buffer[CHUNK_SIZE:]

    # 保存最后一批不足一个 Chunk 的数据
    if len(atom_buffer) > 0:
        chunk_count += 1
        chunk_filename = os.path.join(OUTPUT_DIR, f"chunk_{chunk_count:04d}_atoms.pt")
        torch.save(atom_buffer, chunk_filename)
        save_progress(len(dataset) - 1, chunk_count)

    pbar.close()
    print(f"Extraction finished. Total atom chunks: {chunk_count}")

if __name__ == "__main__":
    main()

import os
import sys
import json
import torch
import numpy as np
from tqdm import tqdm
from rdkit import Chem

# 确保能找到 utils 及其内部的 prompt_vocab
current_dir = os.getcwd()
sys.path.insert(0, current_dir)

from utils.dataset import ZincLMDBDataset
from utils.build_mol import MoleculeBuilder
import utils.mol_descriptor_rich as desc_engine

# ═══════════════════════════════════════════════════════════════════════════
# 🎛️ 配置中心 (与 mol_descriptor_rich 保持同步)
# ═══════════════════════════════════════════════════════════════════════════

class Config:
    # 路径配置
    LMDB_PATH = "/home/course_project/MolFLAE/Latent_Experiments/data/zinc9m/ZINC_9M.lmdb"
    INDICES_PATH = "/home/course_project/MolFLAE/Latent_Experiments/data/zinc9m/dataset_indices.npy"
    OUTPUT_DIR = "/home/course_project/MolFLAE/Latent_Experiments/data/zinc9m_desc/"
    PROGRESS_FILE = os.path.join(OUTPUT_DIR, "metadata.json")

    # 处理参数 (必须与 latent 生成脚本一致)
    CHUNK_SIZE = 81920
    N_VARIANTS = 10            
    SEED = 42                    # 随机种子基准

    # 语言风格控制 (复用 rich 脚本的配置逻辑)
    LANGUAGE_RICHNESS = 0.8
    GRANULARITY = 0.6
    PROB_EXACT_STRUCTURAL_COUNT = 0.25

# ═══════════════════════════════════════════════════════════════════════════
# 核心处理流程
# ═══════════════════════════════════════════════════════════════════════════

def load_progress():
    if os.path.exists(Config.PROGRESS_FILE):
        with open(Config.PROGRESS_FILE, 'r') as f:
            return json.load(f)
    return {"last_processed_idx": -1, "chunk_count": 0}

def save_progress(idx, chunk_count):
    with open(Config.PROGRESS_FILE, 'w') as f:
        json.dump({"last_processed_idx": idx, "chunk_count": chunk_count}, f)

def main():
    os.makedirs(Config.OUTPUT_DIR, exist_ok=True)
    
    # 1. 初始化数据集 (严格不 Shuffle，保证顺序一致)
    print("Initializing Dataset...")
    dataset = ZincLMDBDataset(Config.LMDB_PATH, Config.INDICES_PATH, split='train')
    
    # 2. 加载进度
    progress = load_progress()
    start_idx = progress["last_processed_idx"] + 1
    chunk_count = progress["chunk_count"]
    
    if start_idx >= len(dataset):
        print("All molecules already processed.")
        return

    print(f"Resuming from index: {start_idx}, Chunk Count: {chunk_count}")

    # 3. 初始化分子构建器
    builder = MoleculeBuilder()
    
    # 同步 mol_descriptor_rich 的配置参数
    desc_engine.Config.N_VARIANTS = Config.N_VARIANTS
    desc_engine.Config.LANGUAGE_RICHNESS = Config.LANGUAGE_RICHNESS
    desc_engine.Config.GRANULARITY = Config.GRANULARITY
    desc_engine.Config.PROB_EXACT_STRUCTURAL_COUNT = Config.PROB_EXACT_STRUCTURAL_COUNT

    # 4. 迭代处理
    desc_buffer = []
    
    # 既然需要严格顺序，我们直接按索引遍历，不使用复杂的 DataLoader 以免多进程导致顺序错乱风险
    pbar = tqdm(total=len(dataset), initial=start_idx, desc="Generating Descriptions")

    for i in range(start_idx, len(dataset)):
        try:
            # 获取数据条目
            data = dataset[i]
            mol_id = dataset.ids[i]
            
            # 将 Data 对象还原为 RDKit Molecule (由坐标和原子序数推断键)
            coords = data.x.numpy()
            atoms = data.h.numpy().tolist()
            mol = builder.build_mol(coords, atoms)
            
            if mol is not None:
                # ══════════════ 核心修复代码 ══════════════
                try:
                    # 1. 更新属性缓存
                    mol.UpdatePropertyCache(strict=False)
                    # 2. 强制识别环信息 (解决 RingInfo not initialized 报错)
                    Chem.FastFindRings(mol)
                    # 3. 尝试进行基本的 Sanitization (识别芳香性等，对 LogP 很重要)
                    # 使用较轻量的 sanitize 操作，避免因为化合价问题直接报错
                    Chem.SanitizeMol(mol, Chem.SANITIZE_SYMMRINGS | Chem.SANITIZE_SETAROMATICITY | Chem.SANITIZE_SETCONJUGATION | Chem.SANITIZE_SETHYBRIDIZATION)
                except Exception as e:
                    print(f"\n[Warning] Sanitization failed for mol at index {i}: {e}")
                    # 如果 Sanitization 失败，至少保证环信息存在
                    Chem.FastFindRings(mol)
                # ══════════════════════════════════════════
                # 生成描述符变体 (传入唯一的种子保证多样性且可复现)
                prompts = desc_engine.generate_prompt_variants(
                    mol, 
                    n_variants=Config.N_VARIANTS, 
                    seed=Config.SEED + i
                )
            else:
                # 如果分子构建失败，填入空占位符，保持索引对齐
                prompts = ["A molecular structure."] * Config.N_VARIANTS
                print(f"\n[Warning] Failed to build mol at index {i}")

            desc_buffer.append({
                'id': mol_id,
                'prompts': prompts
            })

            # 5. 满一个 Chunk 保存一次
            if len(desc_buffer) >= Config.CHUNK_SIZE:
                chunk_count += 1
                chunk_filename = os.path.join(Config.OUTPUT_DIR, f"chunk_{chunk_count:04d}.pt")
                
                # 保存为 .pt 格式，与 latent 脚本保持一致
                torch.save(desc_buffer, chunk_filename)
                
                # 更新断点
                save_progress(i, chunk_count)
                
                # 清空缓冲区
                desc_buffer = []

            pbar.update(1)

        except Exception as e:
            print(f"\n[Error] Critical error at index {i}: {e}")
            # 即使报错也要填入占位符，否则会导致后续所有索引错位
            desc_buffer.append({
                'id': dataset.ids[i],
                'prompts': ["Error generating description."] * Config.N_VARIANTS
            })
            pbar.update(1)
            continue

    # 保存最后的残余数据
    if desc_buffer:
        chunk_count += 1
        chunk_filename = os.path.join(Config.OUTPUT_DIR, f"chunk_{chunk_count:04d}.pt")
        torch.save(desc_buffer, chunk_filename)
        save_progress(len(dataset) - 1, chunk_count)

    pbar.close()
    print(f"Description generation finished. Total chunks: {chunk_count}")

if __name__ == "__main__":
    main()

import os
import glob
import json
import torch
from sentence_transformers import SentenceTransformer

# ═══════════════════════════════════════════════════════════════════════════
# 🎛️ 配置中心
# ═══════════════════════════════════════════════════════════════════════════

class Config:
    # 路径配置
    INPUT_DIR = "/home/course_project/MolFLAE/Latent_Experiments/data/zinc9m_desc/"
    OUTPUT_DIR = "/home/course_project/MolFLAE/Latent_Experiments/data/zinc9m_BERT_emb/"
    PROGRESS_FILE = os.path.join(OUTPUT_DIR, "emb_metadata.json")

    # 模型配置
    MODEL_NAME = 'sentence-transformers/all-MiniLM-L6-v2'
    
    # 处理参数
    MAX_CHUNKS_TO_PROCESS = 30  # 只处理前 30 个 Chunk
    N_VARIANTS = 10             # 每个分子的描述符数量 (必须与生成脚本一致，用于 reshape 对齐)
    
    # 编码 Batch Size 
    # (SentenceTransformer 内部的 batch 处理大小，根据你的显存调节，默认 1024 已经很快了)
    BATCH_SIZE = 4096           

# ═══════════════════════════════════════════════════════════════════════════
# 核心处理流程
# ═══════════════════════════════════════════════════════════════════════════

def load_progress():
    """加载断点进度，返回已成功处理的 chunk_filename 列表"""
    if os.path.exists(Config.PROGRESS_FILE):
        with open(Config.PROGRESS_FILE, 'r') as f:
            return json.load(f).get("processed_chunks", [])
    return []

def save_progress(processed_chunks):
    """保存断点进度"""
    with open(Config.PROGRESS_FILE, 'w') as f:
        json.dump({"processed_chunks": processed_chunks}, f)

def main():
    # 1. 初始化目录
    os.makedirs(Config.OUTPUT_DIR, exist_ok=True)
    
    # 2. 搜索并排序 Input Chunks
    input_chunks = sorted(glob.glob(os.path.join(Config.INPUT_DIR, "chunk_*.pt")))
    
    if not input_chunks:
        print(f"❌ No chunk files found in {Config.INPUT_DIR}")
        return

    # 截取前 30 个 Chunk
    target_chunks = input_chunks[:Config.MAX_CHUNKS_TO_PROCESS]
    print(f"🔍 Found {len(input_chunks)} chunks, targeting first {len(target_chunks)} chunks.")

    # 3. 加载进度
    processed_chunks = load_progress()
    print(f"✅ Already processed {len(processed_chunks)} chunks.")

    # 4. 加载模型 (自动检测 GPU 并移动)
    print(f"\n🚀 Loading SentenceTransformer model: {Config.MODEL_NAME} ...")
    model = SentenceTransformer(Config.MODEL_NAME)
    device = "cuda" if torch.cuda.is_available() else "cpu"
    print(f"💻 Model loaded on device: {device}")

    # 5. 遍历处理 Chunks
    for chunk_path in target_chunks:
        chunk_filename = os.path.basename(chunk_path)
        
        # 检查是否已处理
        if chunk_filename in processed_chunks:
            print(f"⏭️  Skipping {chunk_filename} (Already processed)")
            continue

        print(f"\n🔄 Processing {chunk_filename} ...")
        
        # 定义输出路径
        # 例如: chunk_0001.pt -> emb_chunk_0001.pt
        output_filename = f"emb_{chunk_filename}"
        output_path = os.path.join(Config.OUTPUT_DIR, output_filename)

        # 加载文本数据
        # 数据格式: [{'id': mol_id, 'prompts': [p1, p2, ... p10]}, ...]
        data = torch.load(chunk_path)
        num_mols = len(data)
        print(f"   -> Loaded {num_mols} molecules.")

        # =======================================================
        # 高效并行编码策略: Flatten -> Encode -> Reshape
        # =======================================================
        
        # A. 提取并展平所有句子
        # 保证展平后的长度一定是 num_mols * N_VARIANTS
        all_prompts = []
        for item in data:
            if len(item['prompts']) != Config.N_VARIANTS:
                raise ValueError(f"Molecule {item['id']} has {len(item['prompts'])} prompts, expected {Config.N_VARIANTS}!")
            all_prompts.extend(item['prompts'])
        
        total_sentences = len(all_prompts)
        print(f"   -> Encoding {total_sentences} sentences in batches of {Config.BATCH_SIZE}...")

        # B. 批量编码 (返回 PyTorch Tensor 以便直接保存)
        with torch.no_grad():
            embeddings = model.encode(
                all_prompts,
                batch_size=Config.BATCH_SIZE,
                convert_to_tensor=True,      # 直接输出 Tensor
                show_progress_bar=True       # 显示句子编码进度条
            )
        
        # 将数据从 GPU 拉回 CPU 以节省显存，并准备打包
        embeddings = embeddings.cpu()
        emb_dim = embeddings.shape[-1]
        
        # C. 瞬间重构矩阵，完全对齐分子
        # 形状变换: [Total_Sentences, Dim] -> [Num_Mols, N_VARIANTS, Dim]
        emb_reshaped = embeddings.view(num_mols, Config.N_VARIANTS, emb_dim)

        # D. 重新组装列表结构以保存
        out_data = []
        for i, item in enumerate(data):
            out_data.append({
                'id': item['id'],
                # 使用 .clone() 确保切片在内存中连续，减小文件体积
                'embeddings': emb_reshaped[i].clone() 
            })

        # =======================================================

        # 6. 保存 Embedding 数据
        torch.save(out_data, output_path)
        print(f"💾 Saved embeddings to {output_filename} (Shape per mol: {Config.N_VARIANTS}x{emb_dim})")

        # 7. 更新进度并持久化
        processed_chunks.append(chunk_filename)
        save_progress(processed_chunks)

    print("\n🎉 All target chunks processed successfully!")

if __name__ == "__main__":
    main()

from model.train_loop import TrainLoop, center_pos
from config.config import load_config
from utils.dataset import ZincLMDBDataset

# 常量定义
MAP_ATOM_TYPE_ONLY_TO_INDEX = {6: 0, 7: 1, 8: 2, 9: 3, 15: 4, 16: 5, 17: 6, 35: 7, 53: 8}
CHUNK_SIZE = 81920  # 每 81920 个分子保存一个文件
BATCH_SIZE = 5120    # 显存占用考虑
OUTPUT_DIR = "/home/course_project/MolFLAE/Latent_Experiments/data/zinc9m_latent/"
PROGRESS_FILE = os.path.join(OUTPUT_DIR, "metadata.json")

def load_progress():
    if os.path.exists(PROGRESS_FILE):
        with open(PROGRESS_FILE, 'r') as f:
            return json.load(f)
    return {"last_processed_idx": -1, "chunk_count": 0}

def save_progress(idx, chunk_count):
    with open(PROGRESS_FILE, 'w') as f:
        json.dump({"last_processed_idx": idx, "chunk_count": chunk_count}, f)

def main():
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    os.makedirs(OUTPUT_DIR, exist_ok=True)

    # 1. 加载模型
    print("Loading Model...")
    config_path = './config.yaml'
    ckpt_path = 'ckpt-zinc9M/model-epoch=24-val_loss=3.40.ckpt'
    
    config = load_config(config_path)
    model = TrainLoop(config)
    model.load_state_dict(torch.load(ckpt_path, map_location='cpu')['state_dict'])
    model.to(device)
    model.eval()

    # 2. 加载数据集
    print("Loading Training Dataset...")
    lmdb_path = "/home/course_project/MolFLAE/Latent_Experiments/data/zinc9m/ZINC_9M.lmdb"
    indices_path = "/home/course_project/MolFLAE/Latent_Experiments/data/zinc9m/dataset_indices.npy"
    dataset = ZincLMDBDataset(lmdb_path, indices_path, split='train')
    
    # 获取断点信息
    progress = load_progress()
    start_idx = progress["last_processed_idx"] + 1
    chunk_count = progress["chunk_count"]
    
    if start_idx >= len(dataset):
        print("All molecules already processed.")
        return

    print(f"Resuming from index: {start_idx}, Current Chunk Count: {chunk_count}")

    # 使用 Subset 来实现从断点开始
    subset_indices = list(range(start_idx, len(dataset)))
    subset_dataset = torch.utils.data.Subset(dataset, subset_indices)
    
    loader = DataLoader(
        subset_dataset, 
        batch_size=BATCH_SIZE, 
        shuffle=False, 
        num_workers=8, 
        pin_memory=True
    )

    # 3. 循环处理
    latent_buffer = []
    current_idx_in_total = start_idx
    error_batches = []

    pbar = tqdm(total=len(dataset), initial=start_idx, desc="Encoding Latent")
    
    K = model.cfg['encoder_config']['ligand_v_dim']

    # --- 修改 generate_latent_dataset.py 的循环部分 ---

    for batch in loader:
        num_mols_in_batch = batch.batch.max().item() + 1
        batch_mol_ids = dataset.ids[current_idx_in_total : current_idx_in_total + num_mols_in_batch]
        
        try:
            batch = batch.to(device)
            h_idx = torch.tensor([MAP_ATOM_TYPE_ONLY_TO_INDEX.get(a.item(), 0) for a in batch.h], device=device)
            one_hot_h = F.one_hot(h_idx, K).float()
            x_centered, _ = center_pos(batch.x, batch.batch, mode=True)

            with torch.no_grad():
                Zh, Zx, global_batch, _, _ = model.encode(one_hot_h, x_centered, batch.batch, deterministic=True)

            # ================= 修复核心逻辑 =================
            # 此时 Zh 形状为 (B * 10, 32), Zx 形状为 (B * 10, 3)
            # 我们需要将其变形为 (B, 10, 32) 和 (B, 10, 3)
            Zh = Zh.view(num_mols_in_batch, 10, -1)
            Zx = Zx.view(num_mols_in_batch, 10, -1)

            # 正常收集
            for i in range(num_mols_in_batch):
                latent_buffer.append({
                    'id': batch_mol_ids[i],
                    'Zh': Zh[i].cpu(),  # 现在 Zh[i] 是 (10, 32) 的正确张量了！
                    'Zx': Zx[i].cpu()   # Zx[i] 是 (10, 3)
                })
            # ===============================================

        except Exception as e:
            print(f"\n[Error] Batch starting at {current_idx_in_total} failed: {e}")
            # ================= 占位符也必须是 10 个节点 =================
            for i in range(num_mols_in_batch):
                latent_buffer.append({
                    'id': batch_mol_ids[i],
                    'Zh': torch.zeros(10, config['optimal_layer_config']['latent_dim']), 
                    'Zx': torch.zeros(10, 3) 
                })
            # ==========================================================
            error_batches.append((current_idx_in_total, str(e)))

        current_idx_in_total += num_mols_in_batch
        pbar.update(num_mols_in_batch)

        # 4. 检查是否满一个 Chunk 并保存
        if len(latent_buffer) >= CHUNK_SIZE:
            chunk_count += 1
            chunk_filename = os.path.join(OUTPUT_DIR, f"chunk_{chunk_count:04d}.pt")
            
            # 只保存需要的部分，减小体积
            torch.save(latent_buffer[:CHUNK_SIZE], chunk_filename)
            
            # 更新断点（注意：记录的是已经存入文件的最后一个分子的绝对索引）
            save_progress(current_idx_in_total - (len(latent_buffer) - CHUNK_SIZE) - 1, chunk_count)
            
            # 移除已保存部分
            latent_buffer = latent_buffer[CHUNK_SIZE:]


    # 保存最后一批不足一个 Chunk 的数据
    if len(latent_buffer) > 0:
        chunk_count += 1
        chunk_filename = os.path.join(OUTPUT_DIR, f"chunk_{chunk_count:04d}.pt")
        torch.save(latent_buffer, chunk_filename)
        save_progress(len(dataset) - 1, chunk_count)

    pbar.close()
    print(f"Encoding finished. Total chunks: {chunk_count}")

if __name__ == "__main__":
    main()
